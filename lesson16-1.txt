
#import <Foundation/Foundation.h>
@interface NSObject (Random)
+(int)ran: (int) n;
@end;

@implementation NSObject (Random)
+(int)ran: (int) n{
  static int a = 3;
  static int a1 =1;
  a1 = a1+3;
  a = a+a+113*a1;
  int ran = (a*13)%n >= 0 ? (a*13)%n : -(a*(13))%n;

  return ran ;
}   
@end;
